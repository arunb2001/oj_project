#include<bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    cin >> t;
    while(t--) {
        int n, x, total=0;
        cin >> n;
        for(int i=0; i<n; i++) {
            cin >> x;
            total += x;
        }
        cout << total+12 << "\n";
    }
    return 0;
}
